ALTER TABLE ITENSRECEITA ADD TOTAL_ITEM NUMERIC(10,2)

SELECT i.ID_RECEITA, SUM (i.TOTAL_ITEM) AS total
	FROM ITENSRECEITA i
	GROUP BY i.ID_RECEITA 
	
	
UPDATE ITENSRECEITA I
SET TOTAL_ITEM = (SELECT M.PRECO * I.QUANTIDADE FROM MEDICAMENTOS M
	WHERE M.ID_MEDICAMENTO = I.ID_MEDICAMENTO)
